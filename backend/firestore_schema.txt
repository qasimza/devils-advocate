firestore/
  sessions/
    {session_id}/
      # Session metadata
      created_at: timestamp
      ended_at: timestamp
      duration_seconds: number
      consent_given: true          # only exists if user opted in
      
      # The idea being debated
      user_claim: string
      
      # Conversation
      turns: [
        {
          speaker: "user" | "agent"
          text: string
          timestamp: timestamp
          turn_index: number
        }
      ]
      
      # Claim tracking
      claim_events: [
        {
          classification: "DEFENDED" | "CONCEDED" | "NEW_CLAIM" | "DEFLECTED"
          summary: string
          strength: number          # 1-10
          turn_number: number
          timestamp: timestamp
        }
      ]
      
      # End report
      report: {
        overall_score: number
        verdict: string
        strengths: string[]
        weaknesses: string[]
        sharpest_moment: string
        biggest_gap: string
        recommendation: string
      }
      
      # Engagement metrics
      metrics: {
        total_turns: number
        user_turns: number
        agent_turns: number
        interruption_count: number
        defended_count: number
        conceded_count: number
        avg_argument_strength: number
      }
